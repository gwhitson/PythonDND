CREATE TABLE entities (id INTEGER PRIMARY KEY ASC, name TEXT, role TEXT, hp INTEGER, ac INTEGER, grid_x INTERGER, grid_y INTERGER);
CREATE TABLE attacks (id INTEGER PRIMARY KEY ASC, name TEXT, damage TEXT, range INTEGER, poss_player INTEGER, FOREIGN KEY (poss_player) REFERENCES 'entities'(id));
CREATE TABLE game (player_id INTEGER, grid_x INTEGER, grid_y INTEGER, initiative INTEGER, targetted INTEGER, current_atk INTEGER, FOREIGN KEY (player_id) REFERENCES entities(id), FOREIGN KEY (current_atk) REFERENCES attacks(id));
PRAGMA foreign_keys=on;
