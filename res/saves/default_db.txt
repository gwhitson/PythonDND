CREATE TABLE entities (id INTEGER PRIMARY KEY ASC, name TEXT, role TEXT, hp INTEGER, ac INTEGER, move_spd INTEGER not null, grid_x INTEGER not null, grid_y INTEGER not null, pix_x INTEGER not null, pix_y INTEGER not null, sprite TEXT) STRICT;
CREATE TABLE attacks (id INTEGER PRIMARY KEY ASC, name TEXT, damage TEXT, range INTEGER, poss_player INTEGER, FOREIGN KEY (poss_player) REFERENCES 'entities'(id)) STRICT;

CREATE TABLE game (initiative TEXT, last_ent INTEGER, curr_ent INTEGER, next_ent INTEGER, mode TEXT not null, targetted TEXT, map_size INTEGER not null, flags TEXT, FOREIGN KEY (last_ent) REFERENCES entities(id), FOREIGN KEY (curr_ent) REFERENCES entities(id), FOREIGN KEY (next_ent) REFERENCES entities(id)) STRICT;
